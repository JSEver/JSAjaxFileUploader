Basic Uploader
Example Code:
	<div id="target-div1"></div>
	<script>
		$('#target-div1').JSAjaxFileUploader({
			uploadUrl:'upload.php'
		});
	</script>
Manual Upload Initiate Button
Example Code:
	<div id="target-div1-1"></div>
	<script>
		$('#target-div1').JSAjaxFileUploader({
			uploadUrl:'upload.php',
			autoSubmit:false,	//to disable auto submit
			uploadTest:'Start Upload'
		});
	</script>
Custom file selector Name
Example Code:
	<div id="target-div2"></div>
	<script>
		$('#target-div2').JSAjaxFileUploader({
			uploadUrl:'upload.php',
			inputText:'Pick your file for upload...'
		});
	</script>
Upload with custom file name
Example Code:
	<div id="target-div3"></div>
	<script>
		$('#target-div3').JSAjaxFileUploader({
			uploadUrl:'upload.php',
			fileName:'userFile'
		});
	</script>
Restrict file types and file size
Example Code:
	<div id="target-div4"></div>
	<script>
		$('#target-div4').JSAjaxFileUploader({
			uploadUrl:'upload.php',
			maxFileSize:512000,	//Max 500 KB file
			allowExt: 'gif|jpg|jpeg|png'	//allowing only images for upload
		});
	</script>
Upload with form data
Example Code:
	<div id="target-div5"></div>
	<script>
		$('#target-div5').JSAjaxFileUploader({
			uploadUrl:'upload.php',
			formData:{field1:'value1',field2:'value2',fieldN:'valueN'}
		});
	</script>
Custom Animation Speed
Example Code:
	<div id="target-div6"></div>
	<script>
		$('#target-div6').JSAjaxFileUploader({
			uploadUrl:'upload.php',
			closeAnimationSpeed: 1000 //In milli seconds 1000 ms = 1 second
		});
	</script>
Enable/Disable Enlarged image preview
Example Code:
	<div id="target-div7"></div>
	<script>
		$('#target-div7').JSAjaxFileUploader({
			uploadUrl:'upload.php',
			zoomPreview:true,
			zoomWidth:360,
			zoomHeight:360
		});
	</script>
Advance File Upload Control
Example Code:
	<div id="target-div8"></div>
	<script>
		$('#target-div8').JSAjaxFileUploader({
			uploadUrl:'upload.php',
			beforesend:function(file){$('#result').append('Before Send Executed : File -> '+JSON.stringify(file));},
			always:function(file){$('#result').append('Always Executed : File -> '+JSON.stringify(file));},
			done:function(file,xhr){$('#result').append('Done Executed : File -> '+JSON.stringify(file));},
			fail:function(response){$('#result').append('Fail Send Executed : response -> '+response);},
			complete:function(file,xhr){$('#result').append('Complete Executed : File -> '+JSON.stringify(file));}, 
			success:function(response){$('#result').append('Success Executed : response -> '+response);}, 
			error:function(response){$('#result').append('Error Executed : response -> '+response);}
		});
	</script>
